// Compound Interest Simulator
function setupCompoundInterest(){
  const principalEl = document.getElementById('ci-principal');
  const rateEl = document.getElementById('ci-rate');
  const yearsEl = document.getElementById('ci-years');
  const runBtn = document.getElementById('ci-run');
  const out = document.getElementById('ci-output');

  runBtn.addEventListener('click', ()=>{
    const p = parseFloat(principalEl.value) || 0;
    const r = parseFloat(rateEl.value) / 100 || 0;
    const n = parseInt(yearsEl.value) || 0;
    let results = [];
    for(let i=0;i<=n;i++){
      const value = p * Math.pow(1 + r, i);
      results.push({year:i, value: value});
    }
    out.textContent = JSON.stringify(results, null, 2);
  });
}

// Progress bar animation
function setupProgress(){
  const startBtn = document.getElementById('startProgress');
  const bar = document.getElementById('progressBar');
  startBtn.addEventListener('click', ()=>{
    animateProgress(bar, 0, 100, 3500);
  });
}
function animateProgress(el, from, to, duration){
  const start = performance.now();
  function frame(now){
    const t = Math.min(1, (now - start) / duration);
    const eased = easeOutCubic(t);
    const val = from + (to - from) * eased;
    el.style.width = val + '%';
    if(t < 1) requestAnimationFrame(frame);
  }
  requestAnimationFrame(frame);
}
function easeOutCubic(t){return 1 - Math.pow(1 - t, 3)}

// Hero canvas animation: simple particles + rising bars representing "wealth growth"
function setupHeroCanvas(){
  const canvas = document.getElementById('heroCanvas');
  if(!canvas) return;
  const ctx = canvas.getContext('2d');
  let w = canvas.width = canvas.clientWidth || 800;
  let h = canvas.height = canvas.clientHeight || 450;

  window.addEventListener('resize', ()=>{
    w = canvas.width = canvas.clientWidth || 800;
    h = canvas.height = canvas.clientHeight || 450;
  });

  // generate bars
  const bars = Array.from({length:16}).map((_,i)=>({
    x: (i+1) * (w / 18),
    height: Math.random() * (h * 0.6) + 20,
    target: Math.random() * (h * 0.6) + 20,
    hue: 160 + Math.random() * 80,
    speed: 0.008 + Math.random() * 0.02
  }));

  const particles = [];

  function tick(t){
    ctx.clearRect(0,0,w,h);
    // background subtle gradient
    const g = ctx.createLinearGradient(0,0,0,h);
    g.addColorStop(0,'rgba(12,17,28,1)');
    g.addColorStop(1,'rgba(4,8,12,1)');
    ctx.fillStyle = g;
    ctx.fillRect(0,0,w,h);

    // draw bars
    bars.forEach((b,i)=>{
      // approach target
      b.height += (b.target - b.height) * b.speed;
      if(Math.abs(b.target - b.height) < 0.5){
        if(Math.random() < 0.04) b.target = Math.random() * (h * 0.6) + 20;
      }
      const bw = Math.max(8, w / 22);
      const x = b.x - bw / 2;
      const y = h - b.height;
      // glow
      ctx.fillStyle = `rgba(16,185,129,0.06)`;
      ctx.fillRect(x - 6, y - 8, bw + 12, b.height + 16);

      // main bar
      const grad = ctx.createLinearGradient(0,y,0,h);
      grad.addColorStop(0,`hsla(${b.hue},70%,60%,0.95)`);
      grad.addColorStop(1,`hsla(${b.hue+40},70%,40%,0.85)`);
      ctx.fillStyle = grad;
      roundRect(ctx, x, y, bw, b.height, 6);
      ctx.fill();

      // occasionally spawn particles
      if(Math.random() < 0.08){
        particles.push({x: x + bw/2, y: y + Math.random()*10, vx: (Math.random()-0.5)*1.2, vy: -1.5 - Math.random()*1.4, life: 60 + Math.random()*80, hue: b.hue});
      }
    });

    // particles
    for(let i=particles.length-1;i>=0;i--){
      const p = particles[i];
      p.x += p.vx; p.y += p.vy; p.vy += 0.03; p.life--;
      ctx.globalCompositeOperation = 'lighter';
      ctx.fillStyle = `hsla(${p.hue},80%,65%,${Math.max(0, p.life/120)})`;
      ctx.beginPath();
      ctx.arc(p.x, p.y, 2 + Math.random()*2, 0, Math.PI*2);
      ctx.fill();
      if(p.life <= 0) particles.splice(i,1);
    }
    ctx.globalCompositeOperation = 'source-over';

    requestAnimationFrame(tick);
  }
  requestAnimationFrame(tick);
}

function roundRect(ctx, x, y, w, h, r){
  ctx.beginPath();
  ctx.moveTo(x + r, y);
  ctx.arcTo(x + w, y, x + w, y + h, r);
  ctx.arcTo(x + w, y + h, x, y + h, r);
  ctx.arcTo(x, y + h, x, y, r);
  ctx.arcTo(x, y, x + w, y, r);
  ctx.closePath();
}

/*
 * AnimationApp.java
 * A JavaFX application that shows an animated "financial dashboard" style set
 * of animated bars, floating coins, and an interactive slider to control speed.
 */

import javafx.animation.AnimationTimer;
import javafx.animation.KeyFrame;
import javafx.animation.Timeline;
import javafx.application.Application;
import javafx.beans.property.DoubleProperty;
import javafx.beans.property.SimpleDoubleProperty;
import javafx.geometry.Insets;
import javafx.scene.Group;
import javafx.scene.Scene;
import javafx.scene.canvas.Canvas;
import javafx.scene.canvas.GraphicsContext;
import javafx.scene.control.Button;
import javafx.scene.control.Label;
import javafx.scene.control.Slider;
import javafx.scene.layout.BorderPane;
import javafx.scene.layout.HBox;
import javafx.scene.layout.VBox;
import javafx.scene.paint.Color;
import javafx.scene.text.Font;
import javafx.stage.Stage;
import javafx.util.Duration;

public class AnimationApp extends Application {

    private static final int W = 1000;
    private static final int H = 600;
    private double[] barHeights;
    private double[] barTargets;
    private double[] barSpeeds;
    private double[] barX;
    private DoubleProperty speedMultiplier = new SimpleDoubleProperty(1.0);

    @Override
    public void start(Stage stage) {
        stage.setTitle("Elipath Academy ‚Äî JavaFX Animation Preview");

        BorderPane root = new BorderPane();
        Canvas canvas = new Canvas(W, H);
        GraphicsContext gc = canvas.getGraphicsContext2D();

        VBox controls = new VBox(10);
        controls.setPadding(new Insets(10));
        Label lbl = new Label("Animation Speed");
        Slider slider = new Slider(0.2, 3.0, 1.0);
        slider.setShowTickLabels(true);
        slider.setShowTickMarks(true);
        speedMultiplier.bind(slider.valueProperty());

        Button randomize = new Button("Randomize Targets");
        randomize.setOnAction(e -> randomizeTargets());

        controls.getChildren().addAll(lbl, slider, randomize);

        root.setRight(controls);
        root.setCenter(new Group(canvas));

        Scene scene = new Scene(root, W + 240, H, Color.web("#071024"));
        stage.setScene(scene);
        stage.show();

        initializeBars();

        AnimationTimer timer = new AnimationTimer() {
            private long last = 0;
            @Override
            public void handle(long now) {
                if (last == 0) last = now;
                double dt = (now - last) / 1_000_000_000.0; // seconds
                dt *= speedMultiplier.get();
                last = now;
                update(dt);
                render(gc);
            }
        };
        timer.start();
    }

    private void initializeBars() {
        int count = 14;
        barHeights = new double[count];
        barTargets = new double[count];
        barSpeeds = new double[count];
        barX = new double[count];
        for (int i = 0; i < count; i++) {
            barHeights[i] = Math.random() * 150 + 20;
            barTargets[i] = Math.random() * 300 + 50;
            barSpeeds[i] = 0.5 + Math.random() * 1.5;
            barX[i] = 60 + i * ( (W - 120) / (count) );
        }
    }

    private void randomizeTargets() {
        for (int i = 0; i < barTargets.length; i++) {
            barTargets[i] = Math.random() * 380 + 20;
        }
    }

    private void update(double dt) {
        for (int i = 0; i < barHeights.length; i++) {
            double diff = barTargets[i] - barHeights[i];
            barHeights[i] += diff * 0.5 * dt * barSpeeds[i];
            // occasionally nudge targets
            if (Math.random() < 0.005) barTargets[i] = Math.random() * 380 + 20;
        }
    }

    private void render(GraphicsContext gc) {
        // clear
        gc.setFill(Color.web("#071024"));
        gc.fillRect(0, 0, W, H);

        // draw grid lines
        gc.setStroke(Color.rgb(255,255,255,0.04));
        for (int y = 50; y < H; y += 50) {
            gc.strokeLine(40, y, W - 40, y);
        }

        // draw bars with glow
        for (int i = 0; i < barHeights.length; i++) {
            double x = barX[i];
            double bw = (W - 120) / barHeights.length * 0.55;
            double h = barHeights[i];
            double y = H - 80 - h;
            Color base = Color.hsb(160 + i * 8, 0.7, 0.9, 1.0);

            gc.setGlobalAlpha(0.12);
            gc.setFill(base);
            gc.fillRoundRect(x - 8, y - 12, bw + 16, h + 24, 20, 20);
            gc.setGlobalAlpha(1.0);

            gc.setFill(base.darker());
            gc.fillRoundRect(x, y, bw, h, 8, 8);

            gc.setFill(Color.WHITE);
            gc.setFont(Font.font(11));
            gc.fillText(String.format("¬£%dK", (int) (h/10)), x, y - 8);
        }

        // draw title
        gc.setFill(Color.web("#e6eef6"));
        gc.setFont(Font.font(22));
        gc.fillText("Elipath Academy ‚Äî Live Preview", 50, 36);

        // draw footer
        gc.setFill(Color.web("#9aa7b2"));
        gc.setFont(Font.font(12));
        gc.fillText("Interactive JavaFX animation ‚Äî use slider to adjust speed.", 50, H - 24);
    }

    public static void main(String[] args) {
        launch(args);
    }
}
